```js
let _predId = 1;
const makePrediction = (infoId) => {
  const info = PORT_INFO.find(i => i.id === infoId);
  const resources = Object.keys(RESOURCES);
  const portKeys  = Object.keys(PORTS);
  const resource  = resources[Math.floor(Math.random() * resources.length)];
  const targetPort= portKeys[Math.floor(Math.random() * portKeys.length)];
  const direction = Math.random() < 0.5 ? 'up' : 'down';
  // 폭: rumor=방향만, hint=±10~25%, analysis=±15~35%, report=±20~45%
  const magRanges = { rumor:[0,0], hint:[10,25], analysis:[15,35], report:[20,45] };
  const [lo, hi] = magRanges[infoId] || [0,0];
  const mag = lo === 0 ? 0 : Math.floor(lo + Math.random() * (hi - lo));
  // 시점: rumor=모름, hint=1~3시세, analysis=1~2시세, report=1시세
  const turnsRanges = { rumor:[1,3], hint:[1,3], analysis:[1,2], report:[1,1] };
  const [tlo, thi] = turnsRanges[infoId] || [1,3];
  const turnsUntil = Math.floor(tlo + Math.random() * (thi - tlo + 1));
  return {
    id: _predId++, infoId,
    accuracy: info?.accuracy || 0.4,
    resource, targetPort,
    targetPortName: PORTS[targetPort]?.name || targetPort,
    direction, mag,
    turnsUntil, turnsRemaining: turnsUntil,
    applied: false, hit: null,
    showMag: infoId !== 'rumor',      // rumor는 방향만 표시
    showTurns: infoId === 'analysis' || infoId === 'report',
  };
};
```

```js
const rollEventForDeparture = (ship, travelHours, crew, defPct) => {
  // 항로당 최대 1회, 순풍은 별도 판정
  const baseProb = Math.min(0.80, travelHours * 0.08);
  const adjustedProb = Math.max(0, baseProb - (defPct || 0) / 100);
  if (Math.random() > adjustedProb) return null; // 이벤트 없음

  // 순풍은 별도 10% 확률
  if (Math.random() < 0.10) {
    return EVENT_TABLE.find(e => e.id === 'tailwind');
  }
  // 나머지 이벤트 중 랜덤
  const nonTailwind = EVENT_TABLE.filter(e => e.id !== 'tailwind');
  return nonTailwind[Math.floor(Math.random() * nonTailwind.length)];
};

// 출발 핸들러 내부에서:
const stats = calcStats(ship, gs.crew);
const travelDist = Math.hypot(targetX - ship.x, targetY - ship.y);
const travelHours = travelDist / (stats.speed * 3600); // 대략적 시간 계산
const rolledEvent = rollEventForDeparture(ship, travelHours, gs.crew, stats.defPct);

if (rolledEvent) {
  const expiresAt = Date.now() + travelHours * 3600 * 1000 + 3600 * 1000; // 도착 후 1시간
  if (rolledEvent.autoResolve) {
    // 순풍: 자동 처리
    setLog(l => [`💨 ${ship.name || SHIP_TYPES[ship.type].name}: 순풍! 항해 시간 -30%`, ...l]);
    // 도착 시간 단축은 이동 속도에 반영 (ship.booster 필드 재활용 또는 windBonus 추가)
  } else {
    // 일반 이벤트: pendingEvents 큐에 추가
    const pending = {
      id: `evt_${Date.now()}_${ship.id}`,
      shipId: ship.id,
      shipName: ship.name || SHIP_TYPES[ship.type].name,
      eventId: rolledEvent.id,
      expiresAt,
      autoResolved: false,
      autoChoiceId: null,
    };
    setGs(prev => ({
      ...prev,
      pendingEvents: [...prev.pendingEvents, pending],
      ships: prev.ships.map(s => s.id === ship.id
        ? { ...s, travelExtraHours: (s.travelExtraHours || 0) + 1 } // +1h
        : s
      ),
    }));
    setLog(l => [`${rolledEvent.icon} ${ship.name || SHIP_TYPES[ship.type].name}: ${rolledEvent.name} 발생! (+1시간)`, ...l]);
```

기존 이벤트 UI 코드를 참조하여 새 컴포넌트 작성. 컴포넌트를 App.jsx 내 상수 영역 바로 위에 추가:

```jsx
const EventChoiceOverlay = ({ pendingEvent, onChoice, gs, setGs, setLog }) => {
  const evtDef = EVENT_TABLE.find(e => e.id === pendingEvent.eventId);
  if (!evtDef) return null;
  const timeLeft = Math.max(0, Math.ceil((pendingEvent.expiresAt - Date.now()) / 60000));

  const handleChoice = (choice) => {
    // 효과 적용
    setGs(prev => {
      let next = { ...prev };
      const eff = choice.effect || {};
      if (eff.avoidGems) {
        if (prev.gems < eff.avoidGems) return prev; // 보석 부족 — 선택 불가
        next = { ...next, gems: next.gems - eff.avoidGems };
      }
      if (eff.goldLoss) {
        next = { ...next, gold: Math.max(0, next.gold - eff.goldLoss) };
      }
      if (eff.hullDamage) {
        next = {
          ...next,
          ships: next.ships.map(s => s.id === pendingEvent.shipId
            ? { ...s, hull: Math.max(0, (s.hull ?? 100) - eff.hullDamage) }
            : s
          ),
        };
      }
      if (eff.sellPenalty) {
        next = {
          ...next,
          ships: next.ships.map(s => s.id === pendingEvent.shipId
            ? { ...s, sellPenalty: eff.sellPenalty }
            : s
          ),
        };
      }
      if (eff.cargoLoss) {
        next = {
          ...next,
          ships: next.ships.map(s => {
            if (s.id !== pendingEvent.shipId) return s;
            const newCargo = {};
            Object.entries(s.cargo || {}).forEach(([k, v]) => {
              newCargo[k] = Math.max(0, Math.floor(v * (1 - eff.cargoLoss)));
            });
            return { ...s, cargo: newCargo };
          }),
        };
      }
      if (eff.returnToOrigin) {
        next = {
          ...next,
          ships: next.ships.map(s => s.id === pendingEvent.shipId
            ? { ...s, targetX: s.startX, targetY: s.startY }
            : s
          ),
        };
      }
      // 이벤트 정보 보상
      if (evtDef.infoReward && !eff.avoidGems) {
        const infoPred = makePrediction(evtDef.infoReward);
        next = { ...next, predictions: [infoPred, ...next.predictions] };
      }
      // pendingEvents에서 제거
      next = { ...next, pendingEvents: next.pendingEvents.filter(pe => pe.id !== pendingEvent.id) };
      return next;
    });
    setLog(l => [`${evtDef.icon} ${pendingEvent.shipName}: "${choice.label}" 선택 — ${choice.desc}`, ...l]);
    onChoice();
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/70">
      <div className="bg-[#1a2f4a] border-2 border-yellow-500/50 rounded-2xl p-6 max-w-sm w-full mx-4 shadow-2xl">
        <div className="text-3xl text-center mb-2">{evtDef.icon}</div>
        <h3 className="text-lg font-bold text-white text-center mb-1">{evtDef.name}</h3>
        <p className="text-sm text-gray-400 text-center mb-1">{pendingEvent.shipName}</p>
        <p className="text-xs text-orange-400 text-center mb-4">선택 만료: {timeLeft}분 후</p>
        <div className="flex flex-col gap-2">
          {evtDef.choices.map(choice => {
            const isGemChoice = choice.effect?.avoidGems;
            const canAfford = !isGemChoice || (gs.gems >= choice.effect.avoidGems);
            return (
              <button key={choice.id}
                onClick={() => canAfford && handleChoice(choice)}
                className={`p-3 rounded-xl text-left transition border ${
                  canAfford
                    ? 'bg-[#243d5e] border-blue-600/40 hover:border-blue-400 hover:bg-[#2d4d72]'
                    : 'bg-gray-800 border-gray-700 opacity-50 cursor-not-allowed'
                }`}>
                <div className="font-semibold text-white text-sm">{choice.label}</div>
                <div className="text-xs text-gray-400">{choice.desc}</div>
                {isGemChoice && !canAfford && (
                  <div className="text-xs text-red-400">💎 보석 부족</div>
                )}
              </button>
            );
          })}
        </div>
      </div>
    </div>
  );
};
```

```js
const XP_TO_NEXT_LEVEL = (level) => level * 500;

const applyXP = (gs, amount) => {
  const newXP = (gs.playerXP || 0) + amount;
  const needed = XP_TO_NEXT_LEVEL(gs.playerLevel || 1);
  if (newXP >= needed) {
    return {
      gs: { ...gs, playerXP: newXP - needed, playerLevel: (gs.playerLevel || 1) + 1, _pendingLevelUp: true },
      leveled: true,
    };
  }
  return { gs: { ...gs, playerXP: newXP }, leveled: false };
};
```
