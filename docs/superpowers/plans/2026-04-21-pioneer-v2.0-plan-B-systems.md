```js
// src/systems/economy.js
import { PORTS, RESOURCE_REGIONS } from '../constants/index.js';
import { CREW_ROLES } from '../constants/index.js';
import { SHIP_TYPES } from '../constants/index.js';

// ── 거래가 계산 ──
export const calcBuyPrice  = (base, tradePct) =>
  Math.max(1, Math.floor(base * 1.4 * Math.max(0.75, 1 - tradePct / 100)));

export const calcSellPrice = (base, tradePct) =>
  Math.max(1, Math.floor(base * 0.8 * Math.min(1.25, 1 + tradePct / 100)));

// ── 세금 계산 ──
// taxLevel: 1~ / shipCount: 현재 보유 배 수
export const calcTax = (taxLevel, shipCount) => {
  const base = 50 * Math.pow(1.3, Math.max(0, taxLevel - 1));
  const shipBonus = shipCount * Math.max(20, Math.floor(20 * Math.pow(1.1, taxLevel - 1)));
  return Math.floor(base + shipBonus);
};

// ── 선원 역할별 효과 계산 (수확 체감) ──
// crewOnShip: Crew[], role: 'navigator'|'merchant'|'cargo'|'engineer'
const stackEffect = (crewOnShip, role) => {
  const members = crewOnShip.filter(c => c.role === role);
  if (!members.length) return 0;
  return 1 - members.reduce((acc, c) => acc * (1 - c.primaryStat), 1);
};

// ── 배 능력치 계산 (선원 역할 반영) ──
export const calcStats = (ship, crew) => {
  const t = SHIP_TYPES[ship.type];
  const crewOnShip = crew.filter(c => c.shipId === ship.id);

  // 기존 항해술/상술 평균 (하위 호환)
  let navSum = 0, trSum = 0;
  crewOnShip.forEach(c => {
    navSum += c.navigation;
    trSum  += c.trading;
  });
  const avgNav = crewOnShip.length ? navSum / crewOnShip.length : 50;
  const avgTr  = crewOnShip.length ? trSum  / crewOnShip.length : 50;

  // 역할별 효과
  const navBoost      = stackEffect(crewOnShip, 'navigator'); // 속도 +%
  const tradeBoost    = stackEffect(crewOnShip, 'merchant');  // 거래 +%
  const capacityBoost = stackEffect(crewOnShip, 'cargo');     // 수납 +%
  const fuelBoost     = stackEffect(crewOnShip, 'engineer');  // 연료 효율 +%

  // 이벤트 발생률 감소: 항해사 1명당 -5%p
  const navigatorCount   = crewOnShip.filter(c => c.role === 'navigator').length;
  const eventRateReduction = Math.min(0.5, navigatorCount * 0.05); // 최대 50%p 감소

  const fuel     = ship.fuel ?? 100;
  const hull     = ship.hull ?? 100;
  const fuelMult = fuel < 30 ? 0.5 : fuel < 60 ? 0.75 : 1.0;
  const hullMult = hull < 30 ? 0.6 : hull < 60 ? 0.8  : 1.0;
  const fuelEfficiency = 1 + fuelBoost; // 연료 효율 배율 (높을수록 연료 적게 소모)

  const totalRepair = crewOnShip.reduce((a, c) => a + (c.repair || 0), 0);
  const repairDiscount = stackEffect(crewOnShip, 'engineer') * 0.1; // 수리비 최대 10% 할인

  return {
    speed:            Math.max(0.0005, t.baseSpeed * (1 + (avgNav - 50) / 200 + ship.upgrades.speed * 0.15) * fuelMult * (1 + navBoost)),
    capacity:         Math.floor((t.baseCapacity + ship.upgrades.cargo * 25) * (1 + capacityBoost)),
    maxCrew:          Math.min(14, t.maxCrew + ship.upgrades.crew),
    tradePct:         Math.round((avgTr - 50) / 2 * hullMult * (1 + tradeBoost)),
    fuelEfficiency,
    repairDiscount,
    eventRateReduction,
    crewCnt:          crewOnShip.length,
    fuelMult, hullMult, totalRepair,
  };
};

// ── 시세 초기화 ──
export const initPrices = (RESOURCES, PORTS, RESOURCE_REGIONS) => {
  const p = {};
  Object.keys(PORTS).forEach(portKey => {
    p[portKey] = {};
    const port = PORTS[portKey];
    Object.keys(RESOURCES).forEach(res => {
      const rr = RESOURCE_REGIONS[res];
      let base = 80 + Math.random() * 120;
      if (rr) {
        if (rr.cheap.includes(port.region))     base *= 0.7;
        if (rr.expensive.includes(port.region)) base *= 1.4;
      }
      p[portKey][res] = Math.max(20, Math.floor(base));
    });
  });
  return p;
};

// ── 시세 갱신 ──
export const updatePrices = (prevPrices, PORTS, RESOURCES, predictions) => {
  const next = {};
  Object.keys(PORTS).forEach(portKey => {
    next[portKey] = {};
    Object.keys(RESOURCES).forEach(res => {
      const cur = prevPrices[portKey]?.[res] ?? 100;
      let delta = (Math.random() - 0.5) * 60;
      // 예측 적용
      predictions.filter(p => !p.applied && p.targetPort === portKey && p.resource === res)
        .forEach(p => {
          if (Math.random() < p.accuracy) {
            delta += p.direction === 'up' ? p.mag : -p.mag;
          }
        });
      next[portKey][res] = Math.max(20, Math.floor(cur + delta));
    });
  });
  return next;
};
```

```js
  },

const TOTAL_WEIGHT = EVENT_TABLE.reduce((s, e) => s + e.weight, 0);

// 이벤트 랜덤 선택
export const rollEvent = () => {
  let r = Math.random() * TOTAL_WEIGHT;
  for (const evt of EVENT_TABLE) {
    r -= evt.weight;
    if (r <= 0) return evt;
  }
  return EVENT_TABLE[0];
};

// 배에 이벤트 발생 여부 판정
// baseRate: 0.15 (15%), eventRateReduction: 항해사 효과 (0~0.5)
export const shouldTriggerEvent = (baseRate, eventRateReduction) => {
  return Math.random() < Math.max(0.02, baseRate - eventRateReduction);
};

// 이벤트 배 적용 (App.jsx에서 호출)
export const applyEventToShip = (ship, event) => event.apply(ship);
```

```jsx
import React from 'react';

export default function EventOverlay({ activeEvent, gems, onAvoid, onAccept }) {
  if (!activeEvent) return null;
  const { event, shipId } = activeEvent;
  const canAvoid = event.gemCost > 0 && gems >= event.gemCost;

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center"
      style={{ background: 'rgba(0,0,0,0.75)' }}>
      <div className="panel-cr p-6 max-w-sm w-full mx-4 animate-fadeInUp text-center">
        <div className="text-5xl mb-3">{event.icon}</div>
        <h2 className="text-xl font-black mb-1" style={{ color: '#f5c842' }}>
          {event.label}
        </h2>
        <p className="text-sm text-gray-300 mb-4">{event.desc({})}</p>

        <div className="flex gap-3 justify-center">
          {event.gemCost > 0 && (
            <button
              className={`btn-cr text-sm ${!canAvoid ? 'opacity-50 cursor-not-allowed' : ''}`}
              onClick={() => canAvoid && onAvoid(shipId, event)}
              disabled={!canAvoid}
            >
              💎 {event.gemCost}개로 회피
            </button>
          )}
          <button
            className="btn-cr-danger text-sm"
            onClick={() => onAccept(shipId, event)}
          >
            감수하기
          </button>
        </div>

        {event.gemCost > 0 && !canAvoid && (
          <p className="text-xs mt-2" style={{ color: '#e74c3c' }}>
            보석 부족 (보유: {gems}개 / 필요: {event.gemCost}개)
          </p>
        )}

    </div>
  );
}
```

```js
// src/systems/quests.js
import { PORTS, RESOURCES } from '../constants/index.js';

const pick = (arr) => arr[Math.floor(Math.random() * arr.length)];

let _questId = 1;

// 퀘스트 생성 — 항구 도착 시 0~2개 랜덤 생성, 최대 3개 동시
export const generateNewQuests = (activeQuests, currentTaxAmount) => {
  if (activeQuests.length >= 3) return [];
  const count = Math.floor(Math.random() * 3); // 0~2개
  if (count === 0) return [];

  const portKeys  = Object.keys(PORTS);
  const resKeys   = Object.keys(RESOURCES);
  const slots     = 3 - activeQuests.length;
  const generated = [];

  const templates = [
    () => {
      const res    = pick(resKeys);
      const tPort  = pick(portKeys);
      const amount = 10 + Math.floor(Math.random() * 21); // 10~30
      const reward = Math.floor(currentTaxAmount * (2 + Math.random()));
      return {
        id: _questId++, type: 'delivery',
        title: `${RESOURCES[res].icon} 긴급 배송`,
        desc: `${res} ${amount}개를 ${PORTS[tPort].name}에 납품`,
        resource: res, targetPort: tPort, targetAmount: amount,
        progress: 0,
        deadlineMs: 2 * 60 * 60 * 1000, // 2시간
        deadlineAt: Date.now() + 2 * 60 * 60 * 1000,
        reward: { gold: reward, gems: 1 },
        penalty: { taxMultiplier: 1 },
        status: 'active',
      };
    },
    () => {
      const visitAmt = 3 + Math.floor(Math.random() * 3); // 3~5
      const reward   = Math.floor(currentTaxAmount * (2.5 + Math.random()));
      return {
        id: _questId++, type: 'visit',
        title: '📍 항구 탐험',
        desc: `새로운 항구 ${visitAmt}곳 방문`,
        targetAmount: visitAmt, visitedPorts: [],
        progress: 0,
        deadlineMs: 4 * 60 * 60 * 1000,
        deadlineAt: Date.now() + 4 * 60 * 60 * 1000,
        reward: { gold: reward, gems: 0, specialCrew: true },
        penalty: { taxMultiplier: 1 },
        status: 'active',
      };
    },
    () => {
      const tradeAmt = 5000 + Math.floor(Math.random() * 15001); // 5000~20000
      const reward   = Math.floor(currentTaxAmount * (2 + Math.random()));
      return {
        id: _questId++, type: 'trade',
        title: '💰 무역왕 도전',
        desc: `총 ${tradeAmt.toLocaleString()}금 판매`,
        targetAmount: tradeAmt, progress: 0,
        deadlineMs: 3 * 60 * 60 * 1000,
        deadlineAt: Date.now() + 3 * 60 * 60 * 1000,
        reward: { gold: reward, gems: 0, taxLevelDelta: -1 },
        penalty: { taxMultiplier: 1 },
        status: 'active',
      };
    },
    () => {
      const res    = pick(resKeys);
      const amount = 20 + Math.floor(Math.random() * 31); // 20~50
      const reward = Math.floor(currentTaxAmount * (2 + Math.random()));
      return {
        id: _questId++, type: 'monopoly',
        title: '🏪 독점 계약',
        desc: `${res} ${amount}개 매입`,
        resource: res, targetAmount: amount, progress: 0,
        deadlineMs: 2 * 60 * 60 * 1000,
        deadlineAt: Date.now() + 2 * 60 * 60 * 1000,
        reward: { gold: reward, gems: 0, priceBonusResource: res },
        penalty: { taxMultiplier: 1 },
        status: 'active',
      };
    },
  ];

  for (let i = 0; i < Math.min(count, slots); i++) {
    generated.push(pick(templates)());
  }
  return generated;
};

// 퀘스트 만료 체크 — 타임아웃된 퀘스트 실패 처리
export const checkQuestExpiry = (quests) => {
  const now = Date.now();
  return quests.map(q => {
    if (q.status !== 'active') return q;
    if (now > q.deadlineAt) return { ...q, status: 'failed' };
    return q;
  });
};

// 퀘스트 진행 업데이트
// type: 'delivery' — portKey 도착 + cargo 확인
// type: 'visit'    — portKey 방문
// type: 'trade'    — 판매 금액 누적
// type: 'monopoly' — 구매 수량 누적
export const updateQuestProgress = (quests, event) => {
  return quests.map(q => {
    if (q.status !== 'active') return q;
    switch (q.type) {
      case 'delivery':
        if (event.type === 'arrive' && event.portKey === q.targetPort) {
          const delivered = event.cargo?.[q.resource] ?? 0;
          const progress  = Math.min(q.targetAmount, q.progress + delivered);
          if (progress >= q.targetAmount) return { ...q, progress, status: 'completed' };
          return { ...q, progress };
        }
        return q;
      case 'visit':
        if (event.type === 'arrive' && !q.visitedPorts.includes(event.portKey)) {
          const visitedPorts = [...q.visitedPorts, event.portKey];
          const progress     = visitedPorts.length;
          if (progress >= q.targetAmount) return { ...q, visitedPorts, progress, status: 'completed' };
          return { ...q, visitedPorts, progress };
        }
        return q;
      case 'trade':
        if (event.type === 'sell') {
          const progress = q.progress + (event.amount ?? 0);
          if (progress >= q.targetAmount) return { ...q, progress, status: 'completed' };
          return { ...q, progress };
        }
        return q;
      case 'monopoly':
        if (event.type === 'buy' && event.resource === q.resource) {
          const progress = q.progress + (event.amount ?? 0);
          if (progress >= q.targetAmount) return { ...q, progress, status: 'completed' };
          return { ...q, progress };
        }
        return q;
      default:
        return q;
    }
  });
};

// 남은 시간 포맷 (ms → "1:42:30")
export const formatTimeLeft = (deadlineAt) => {
  const ms = Math.max(0, deadlineAt - Date.now());
  const h  = Math.floor(ms / 3600000);
  const m  = Math.floor((ms % 3600000) / 60000);
  const s  = Math.floor((ms % 60000) / 1000);
  return `${h}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
};
```
