const makeCrew = (grade = null, originPort = null) => {
  const id = _crewSeed++;
  // 등급 결정
  let resolvedGrade = grade;
  if (!resolvedGrade) {
    const roll = Math.random();
    if (roll < 0.03)      resolvedGrade = 'legendary';
    else if (roll < 0.12) resolvedGrade = 'expert';
    else if (roll < 0.35) resolvedGrade = 'skilled';
    else                  resolvedGrade = 'common';
  }
  const gradeConf = CREW_GRADE_STATS[resolvedGrade];
  // 보유 스탯 랜덤 선택 (statCount개)
  const shuffled = [...STAT_KEYS].sort(() => Math.random() - 0.5);
  const chosenStats = shuffled.slice(0, gradeConf.statCount);
  const stats = { speed: 0, cargo: 0, trade: 0, defense: 0, repair: 0 };
  chosenStats.forEach(k => {
    const [lo, hi] = gradeConf.ranges[k];
    stats[k] = Math.floor(lo + Math.random() * (hi - lo + 1));
  });
  // 출신 항구 보너스 (Task 8 이후 PORT_CREW_BONUS 참조)

  if (originPort && typeof PORT_CREW_BONUS !== 'undefined' && PORT_CREW_BONUS[originPort]) {

    originBonus = { statKey: b.statKey, value: b.value };
    stats[b.statKey] = (stats[b.statKey] || 0) + b.value;
  }
  const gradeHireCost = { common: [200,800], skilled: [800,2000], expert: [2000,5000], legendary: [6000,12000] };
  const [lo, hi] = gradeHireCost[resolvedGrade];
  return {
    id,
    name: CREW_NAMES[id % CREW_NAMES.length],
    grade: resolvedGrade,
    stats,
    originPort: originPort || null,
    originBonus,
    hireCost: Math.floor(lo + Math.random() * (hi - lo)),

  };
};

- [ ] **Step 1: calcStats를 새 스탯 기반으로 교체**

기존 `calcStats` 전체를 아래로 교체:

```js
const calcStats = (s, crew) => {
  const t = SHIP_TYPES[s.type];
  const crewOnShip = crew.filter(c => c.shipId === s.id);
  const fuel = s.fuel ?? 100, hull = s.hull ?? 100;
  const fuelMult = fuel < 30 ? 0.5 : fuel < 60 ? 0.75 : 1.0;
  const hullMult = hull < 30 ? 0.6 : hull < 60 ? 0.8 : 1.0;

  // 스탯 합산 (수확 체감: 1 - (1-x)^n 형태)
  const sumStat = (key) => {
    const effects = crewOnShip.map(c => (c.stats?.[key] || 0) / 100);
    return effects.reduce((acc, e) => 1 - (1 - acc) * (1 - e), 0);
  };
  const speedBonus   = sumStat('speed');    // 0~1 (%)
  const cargoBonus   = crewOnShip.reduce((a, c) => a + (c.stats?.cargo || 0), 0); // 칸 단순합
  const tradeBonus   = sumStat('trade');    // 0~1
  const defenseBonus = sumStat('defense');  // 0~1 (이벤트 확률 감소용)
  const repairBonus  = sumStat('repair');   // 0~1

  return {
    speed:      Math.max(0.0005, t.baseSpeed * (1 + speedBonus + s.upgrades.speed * 0.15) * fuelMult * hullMult),
    capacity:   t.baseCapacity + s.upgrades.cargo * 25 + cargoBonus,
    maxCrew:    Math.min(14, t.maxCrew + s.upgrades.crew),
    tradePct:   Math.round(tradeBonus * 40), // 최대 40% 거래 보너스
    defPct:     Math.round(defenseBonus * 30), // 이벤트 확률 최대 -30%
    repairDisc: Math.round(repairBonus * 40),  // 수리비 최대 -40%
    crewCnt:    crewOnShip.length,
    fuelMult, hullMult,
  };
};
```

  const unlocked = new Set();
  UNLOCK_TIERS.forEach(tier => {

- [ ] **Step 1: 세금 징수 로직에 배 매각 우선 처리 추가**

세금 징수 부분을 찾아 gold 부족 처리를 아래로 교체:

```js
const collectTax = (state) => {
  const tax = Math.round(calcTax(state.ships.length, state.taxLevel) * (1 + (state.taxPenaltyPct || 0) / 100));
  if (state.gold >= tax) {
    return { ...state, gold: state.gold - tax };
  }
  // 금화 부족 — 1단계: 배 강제 매각 (저렴한 순, 1척 남기고)
  let newState = { ...state };
  let needed = tax - newState.gold;
  if (newState.ships.length > 1) {
    const sortedShips = [...newState.ships].sort((a, b) => SHIP_TYPES[a.type].cost - SHIP_TYPES[b.type].cost);
    for (const ship of sortedShips) {
      if (newState.ships.length <= 1) break;
      if (needed <= 0) break;
      const sellValue = Math.floor(SHIP_TYPES[ship.type].cost * 0.5);
      newState = {
        ...newState,
        gold: newState.gold + sellValue,
        ships: newState.ships.filter(s => s.id !== ship.id),
        crew: newState.crew.map(c => c.shipId === ship.id ? { ...c, shipId: null } : c),
      };
      needed -= sellValue;
      // 로그 기록 — 실제 로그 setter는 컴포넌트 내부이므로 반환값으로 처리
      newState._forceSaleLog = (newState._forceSaleLog || []).concat(
        `⚠️ 세금 부족 — ${ship.name || SHIP_TYPES[ship.type].name} 강제 매각 (${sellValue.toLocaleString()}금 수령)`
      );
    }
  }
  // 2단계: 보석 대납
  if (newState.gold < tax && newState.gems > 0) {
    const gemCost = 1;
    newState = { ...newState, gold: newState.gold + tax, gems: newState.gems - gemCost };
    newState._forceSaleLog = (newState._forceSaleLog || []).concat('💎 보석 1개로 세금 대납');
  }
  // 3단계: 미납 누적 (다음 납부 2배)
  if (newState.gold < tax) {
    newState._forceSaleLog = (newState._forceSaleLog || []).concat('❌ 세금 미납 — 다음 납부 시 2배 징수');
    newState = { ...newState, taxPenaltyPct: Math.min(20, (newState.taxPenaltyPct || 0) + 5) };
  } else {
    newState = { ...newState, gold: newState.gold - tax };
  }
  return newState;
};
```
