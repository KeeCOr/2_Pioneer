```js
const handleCrewRefresh = (portKey, useGems = false) => {
  const { goldCost, gemCost } = getRefreshCost(portKey, gs);
  if (useGems) {
    if (gs.gems < gemCost) { setLog(l => ['💎 보석 부족 — 새로고침 불가', ...l]); return; }
  } else {
    if (gs.gold < goldCost) { setLog(l => ['💰 금화 부족 — 새로고침 불가', ...l]); return; }
  }
  const newAvailableCrew = generateAvailableCrew(
    gs.taxLevel,
    portKey,
    useGems ? 6 : 5, // 프리미엄은 1명 추가 (전문가+ 풀 포함)
  );
  // 프리미엄: 전문가+ 등급 보장
  const finalCrew = useGems
    ? newAvailableCrew.map((c, i) => i === 0 ? makeCrew('expert', portKey) : c)
    : newAvailableCrew;
  setGs(prev => ({
    ...prev,
    gold: useGems ? prev.gold : prev.gold - goldCost,
    gems: useGems ? prev.gems - gemCost : prev.gems,
    availableCrew: finalCrew,
    portRefreshCounts: {
      ...prev.portRefreshCounts,
      [portKey]: (prev.portRefreshCounts?.[portKey] || 0) + 1,
    },
  }));
  setLog(l => [`🔄 선원 목록 새로고침 (${useGems ? `💎${gemCost}` : `${goldCost}금`})`, ...l]);
};
```

```js
let _deliveryId = 1;
const generateDeliveryQuests = (fromPortKey, count = 3) => {
  const portKeys = Object.keys(PORTS).filter(k => k !== fromPortKey);
  const resKeys  = Object.keys(RESOURCES);
  const pick = arr => arr[Math.floor(Math.random() * arr.length)];
  return Array.from({ length: count }, () => {
    const toPortKey  = pick(portKeys);
    const resource   = pick(resKeys);
    const cargoSlots = 5 + Math.floor(Math.random() * 20);     // 5~24칸
    const dist = Math.hypot(
      PORTS[fromPortKey].x - PORTS[toPortKey].x,
      PORTS[fromPortKey].y - PORTS[toPortKey].y
    );
    const reward = Math.floor(cargoSlots * 18 * (1 + dist / 50)); // 거리 비례 보상
    return {
      id: _deliveryId++,
      fromPort: fromPortKey,
      toPort: toPortKey,
      toPortName: PORTS[toPortKey].name,
      resource,
      cargoSlots,
      reward,
      status: 'available', // 'available' | 'active' | 'done'
    };
  });
};
```

```js
const acceptDelivery = (delivery, shipId) => {
  const ship = gs.ships.find(s => s.id === shipId);
  if (!ship) return;
  const stats = calcStats(ship, gs.crew);
  const freeSlots = stats.capacity - Object.values(ship.cargo || {}).reduce((a, v) => a + v, 0);
  const reserved  = (ship.deliveryReserved || 0);
  const available = freeSlots - reserved;
  if (available < delivery.cargoSlots) {

    return;
  }
  setGs(prev => ({
    ...prev,
    ships: prev.ships.map(s => s.id === shipId
      ? {
          ...s,
          deliveries: [...(s.deliveries || []), { ...delivery, status: 'active', shipId }],
          deliveryReserved: (s.deliveryReserved || 0) + delivery.cargoSlots,
        }
      : s
    ),
```
